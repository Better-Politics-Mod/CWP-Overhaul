onaction_fiscal_calculation = {
    # Money Supply Calculations
    if = {
        limit = {
            has_variable = money_supply
        }
        change_variable = {
            name = money_supply
            add = money_supply_growth
        }
    }
    if = {
        limit = {
            has_variable = inflation_rate
        }
        change_variable = {
            name = inflation_rate
            add = inflation_growth
        }
    }
    if = {
        limit = {
            inflation_value < 10
            inflation_value > -10
        }
        remove_modifier = deflation
        remove_modifier = inflation
    }
    if = {
        limit = {
            inflation_value >= 10
        }
        remove_modifier = deflation
        remove_modifier = inflation
        add_modifier = {
            name = inflation
            multiplier = inflation_modifier_mult
        }
    }
    if = {
        limit = {
            inflation_value <= -10
        }
        remove_modifier = deflation
        remove_modifier = inflation
        add_modifier = {
            name = deflation
            multiplier = inflation_modifier_mult
        }
    }
}
