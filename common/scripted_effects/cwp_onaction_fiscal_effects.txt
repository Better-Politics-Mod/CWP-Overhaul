onaction_fiscal_calculation = {
	# Money Supply Calculations
	if = {
		limit = {
			has_variable = money_supply
		}
		change_variable = {
			name = money_supply
			add = money_supply_growth
		}
	}
	if = {
		limit = {
			has_variable = inflation_rate
		}
		change_variable = {
			name = inflation_rate
			add = inflation_growth
		}
	}
	if = {
		limit = {
			inflation_value < 10
			inflation_value > -1
		}
		remove_modifier = deflation
		remove_modifier = inflation
	}
	if = {
		limit = {
			inflation_value >= 10
		}
		remove_modifier = deflation
		remove_modifier = inflation
		add_modifier = {
			name = inflation
			multiplier = inflation_modifier_mult
		}
	}
	if = {
		limit = {
			inflation_value <= -1
		}
		remove_modifier = deflation
		remove_modifier = inflation
		add_modifier = {
			name = deflation
			multiplier = inflation_modifier_mult
		}
	}
	if = {
		limit = {
			has_variable = contractionary_fiscal_policy
		}
		remove_modifier = monetary_policy
		add_modifier = {
			name = monetary_policy
			multiplier = -1
		}
	}
	if = {
		limit = {
			has_variable = stable_fiscal_policy
			has_modifier = monetary_policy
		}
		remove_modifier = monetary_policy
	}
	if = {
		limit = {
			has_variable = expansionary_fiscal_policy
		}
		remove_modifier = monetary_policy
		add_modifier = {
			name = monetary_policy
			multiplier = 1
		}
	}
}
