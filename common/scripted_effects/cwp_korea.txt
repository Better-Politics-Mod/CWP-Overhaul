
scripted_effect_korean_country_sort = {

	every_in_global_list = {
		if = {
			limit = {
				not = { exists = var:highest_prestige_communist }
			}
			set_variable = {
				name = highest_prestige_communist
				value = THIS
			}
			else_if = {
				limit = {
					THIS.prestige > var:highest_prestige_communist.prestige
				}
				set_variable = {
					name = highest_prestige_communist
					value = THIS
				}
			}
		}
	}
}
