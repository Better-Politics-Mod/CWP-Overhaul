calc_ig_corruption = {
	if = {
		limit = {
			NOT = {
				has_variable = ig_time_in_government
			}
		}
		set_variable = {
			name = ig_time_in_government
			value = 0
		}
	}
	if = {
		limit = {
			is_in_government = yes
		}
		change_variable = {
			name = ig_time_in_government
			add = 1
		}
	}
	else = {
		change_variable = {
			name = ig_time_in_government
			add = -1
		}
	}
	if = {
		limit = {
			NOT = {
				has_variable = ig_corruption
			}
		}
		set_variable = {
			name = ig_corruption
			value = {
				value = ig_corruption_growth
				multiply = 24				# If we don't have any corruption value yet, we start at current growth * 24
			}
		}
	}
	if = {
		limit = {
			NOT = {
				ig_corruption_growth = 0
			}
		}
		change_variable = {
			name = ig_corruption
			add = ig_corruption_growth
		}
	}
	if = {
		limit = {
			var:ig_corruption > 100			# Max of 100 corruption
		}
		set_variable = {
			name = ig_corruption
			value = 100
		}
	}
	if = {
		limit = {
			var:ig_corruption < 0			# Min of 0 corruption
		}
		set_variable = {
			name = ig_corruption
			value = 0
		}
	}
	# Adds a Purely Visual Modifier

	#remove_modifier = ig_corruption_modifier

	#	limit = {
	#		has_variable = ig_corruption
	#		var:ig_corruption > 0
	#	}
	#	add_modifier = {
	#		name = ig_corruption_modifier
	#		multiplier = var:ig_corruption
	#	}
	#}
}
