je_imf_concessionary_loan = {
	icon = "gfx/interface/icons/cwp_icons/imf.dds"
	should_be_pinned_by_default = yes
	possible = {
		is_target_in_global_variable_list = {
			name = imf_members
			target = THIS
		}
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = not_paying_imf_loans_status
				trigger = {
					var:imf_debt_monthly_percentage <= 0 
				}
			}
			triggered_desc = {
				desc = paying_imf_loans_status
				trigger = {
					var:imf_debt_monthly_percentage >= 0.01 
				}
			}
		}
	}
	immediate = {
		set_variable = {
			name = imf_loan_payment_months_current
			value = 0
		}
	}
	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = imf_loan_payment_months_current
				add = 1
			}
		}
	}
	current_value = {
		value = var:imf_loan_payment_months_current
	}
	goal_add_value = {
		add = imf_application_payment_due_months
	}
	on_timeout = {
	
	}
	complete = {
		custom_tooltip = {
			text = IMF_LOAN_PAID
			var:imf_debt <= 0 
		}
	}
	fail = {
		or = {
			scope:journal_entry = {
				is_goal_complete = yes
			}
			has_modifier = defaulted_on_international_loan
		}
	}
	on_complete = {
		custom_tooltip = {
			text = IMF_LOAN_PAID_CONCLUSION
			imf_loan_conclusion = yes
		}

	}
	on_fail = {
		trigger_event = {
			id = imf_events.3
		}
	}
	progressbar = yes
	weight = 100
}
