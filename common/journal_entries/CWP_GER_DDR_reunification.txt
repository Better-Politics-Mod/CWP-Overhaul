je_german_reunification = {
	icon = "gfx/interface/icons/lens_toolbar_icons/nuclear_weapons.dds"
	should_be_pinned_by_default = yes

	scripted_button = german_reunification_button

	is_shown_when_inactive = {
		OR = {
			AND = {
				exists = c:DDR
				this = c:DDR
			}
			AND = {
				exists = c:GER
				this = c:GER
			}
			c:GER = {is_subject_of = ROOT}
			c:DDR = {is_subject_of = ROOT}
		}
	}

	possible = {
		exists = c:GER
		exists = c:DDR
		OR = {
			OR = { #Both Germanys have the same overlord or one Germany is the overlord of the other one
				c:GER = { overlord = { c:DDR = {is_subject_of = PREV} } }
				c:DDR = { overlord = { c:GER = {is_subject_of = PREV} } }
			}
			AND = {
				c:GER = {is_subject = false}
				c:DDR = {is_subject = false}
			}
		}
	}

	immediate = {
		c:GER = {save_scope_as = scope_reuinification_GER}
		c:DDR = {save_scope_as = scope_reuinification_DDR}
		c:GER = { overlord = {save_scope_as = scope_reuinification_GER_overlord}}
		set_global_variable = {
			name = unification_voting_power_GER_je
			value = c:GER.prestige
		}
		set_global_variable = {
			name = unification_voting_power_DDR_je
			value = c:DDR.prestige
		}
		set_global_variable = {
			name = unification_voting_power_GER_overlord_je
			value = c:GER.overlord.prestige
		}
	}

	on_monthly_pulse = {
		c:GER = { overlord = {save_scope_as = scope_reuinification_GER_overlord}} #In case overlord changes
		set_global_variable = {
			name = unification_voting_power_GER_je
			value = c:GER.prestige
		}
		set_global_variable = {
			name = unification_voting_power_DDR_je
			value = c:DDR.prestige
		}
		set_global_variable = {
			name = unification_voting_power_GER_overlord_je
			value = c:GER.overlord.prestige
		}
	}

	complete = {
		always = false
	}
	timeout = 365
	on_timeout = {
	}
	on_complete = {
	}
	on_timeout = {
	}
	weight = 100
}
