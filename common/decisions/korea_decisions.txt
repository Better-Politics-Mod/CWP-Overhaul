join_korean_conflict_ai_north = {
	is_shown = {
		OR = {
			c:PRC ?= this
			c:RUS ?= this
		}
		is_player = no
		exists = c:NRK
		exists = c:STK
		NOT = {
			has_war_with = c:STK
			is_in_war_together = c:NRK
		}
		NOT = {
			has_variable = joined_korean_conflict
		}
	}
	possible = {
		c:NRK = {
			has_war_with = c:STK
		}
		c:STK.relations:root < relations_threshold:amicable
	}
	when_taken = {
		set_variable = joined_korean_conflict
		if = {
			limit = {
				any_diplomatic_play = {
					initiator = c:NRK
					target = c:STK
					is_war = yes
				}
			}
			every_diplomatic_play = {
				limit = {
					initiator = c:NRK
					target = c:STK
					is_war = yes
				}
				war = {
					add_war_war_support = {
						target = c:NRK
						value = 50
					}
				}
				add_initiator_backers = {
					root
				}
			}
		}
		else = {
			every_diplomatic_play = {
				limit = {
					target = c:NRK
					initiator = c:STK
					is_war = yes
				}
				war = {
					add_war_war_support = {
						target = c:NRK
						value = 50
					}
				}
				add_target_backers = {
					root
				}
			}
		}
		every_diplomatic_play = {
			limit = {
				initiator = c:NRK
				target = c:STK
				is_war = yes
			}
			war = {
				add_war_war_support = {
					target = c:NRK
					value = 50
				}
			}
			add_initiator_backers = {
				root
			}
		}
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				is_at_war = no
			}
			add = 5
		}
		modifier = {
			trigger = {
				any_diplomatic_play = {
					OR = {
						AND = {
							initiator = c:NRK
							target = c:STK
						}
						AND = {
							initiator = c:STK
							target = c:NRK
						}
					}
					is_war = yes
					war = {
						has_war_support = {
							target = c:NRK
							value < 60
						}
					}
				}
			}
			add = 100
		}
		modifier = {
			trigger = {
				c:RUS ?= this
			}
			add = 100
		}
	}
}
