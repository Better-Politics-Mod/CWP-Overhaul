form_nato = {
	is_shown = {
		this = {
			this ?= c:USA
			not = {
				has_global_variable = nato_leader
			}
		}
	}
	possible = {
		country_rank >= rank_value:superpower
	}
	when_taken = {
		trigger_event = nato_formation.1
		set_global_variable = {
			name = nato_leader
			value = ROOT
		}
	}
	ai_chance = {
		base = 0
	}
}

join_nato = {
	is_shown = {
		this = {
			nor = {
				is_target_in_global_variable_list = {
					name = nato_members
					target = THIS
				}
			}
			has_diplomatic_pact = {
				who = c:USA
				type = diplomatic_alignment
			}
		}
	}

	possible = {
		this = {
			nor = {
				has_diplomatic_pact = {
					who = c:RUS
					type = diplomatic_alignment
				}
				is_at_war = no
			}
			has_diplomatic_pact = {
				who = c:USA
				type = diplomatic_alignment
			}
		}
	}

	when_taken = {
		this = {
			set_variable = {
				name = application_progress_nato
				value = 0
			}
			set_variable = {
				name = nato_supporting_list_size
				value = 0
			}
			add_to_global_variable_list = {
				name = nato_application_list
				target = THIS
			}
		}
	}

	ai_chance = { # TODO, does not work
		base = 10
	}
}
