law_sovereign_bureaucrats = {
	group = lawgroup_bureaucracy
	
	icon = "gfx/interface/icons/law_icons/appointed_bureaucrats.dds"
	
	progressiveness = 0
	
	modifier = {
		country_bureaucrats_pol_str_mult = 0.80
		state_tax_capacity_mult = 0.25
	}
	
	possible_political_movements = {
		law_elected_bureaucrats
	}

	on_enact = {
		if = {
			limit = {
				has_law = law_type:law_elected_bureaucrats
			}
			custom_tooltip = bureaucrats_pb_ig_shift_effect_negative_tt
		}
		if = {
			limit = {
				has_law = law_type:law_independent_bureaucrats
			}
			custom_tooltip = urban_aristocrats_int_ig_shift_effect_negative_tt
		}
	}

	pop_support = {
		value = 0
		# Academics like technocracy
		add = {
			desc = "POP_ACADEMICS"			
			if = {
				limit = { 
					is_pop_type = academics
				}
				value = 0.33
			}
		}
		# Bureuacrats like non-hereditary
		add = {
			desc = "POP_BUREAUCRATS"
			if = {
				limit = {
					owner = { has_law = law_type:law_independent_bureaucrats }
					is_pop_type = administrators
				}
			}
			value = 0.25
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}
