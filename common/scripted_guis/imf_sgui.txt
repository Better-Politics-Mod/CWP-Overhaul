is_creditor_nation = {
	is_shown = {
		THIS = {
			is_target_in_global_variable_list = {
				name = imf_creditors
				target = THIS
			}
		}
	}
}

imf_contribution_percent_increase = {
	effect = {
		if = { limit = { var:imf_contribution_percent >= 60 } set_variable = { name = imf_contribution_percent value = 60 }}
		if = { limit = { var:imf_contribution_percent < 60 } change_variable = { name = imf_contribution_percent add = 0.5 }}
	}
}

imf_contribution_percent_decrease = {
	effect = {
		if = { limit = { var:imf_contribution_percent > 0.5 } change_variable = { name = imf_contribution_percent subtract = 0.5}}
		else_if = { limit = { var:imf_contribution_percent <= 0.5 } set_variable = { name = imf_contribution_percent value = 0}}
	}
}
