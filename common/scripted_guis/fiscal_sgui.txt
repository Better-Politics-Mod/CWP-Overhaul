### Foreign Reserves
buy_foreign_currency = {
	effect = {
		# Removes From Treasury
		add_treasury = -500000
		# Increases Reserve
		change_variable = {
			name = foreign_reserves
			add = {
				value = 500000
				multiply = {
					value = ROOT.country_currency_val
					divide = scope:currency_exchange_country.country_currency_val
				}
			}
		}
		# Increases Bought Currency Value
		scope:currency_exchange_country = {
			on_action_change_domestic_currency_value = {
				mathAction = add
				mathAction2 = currency_value_growth_by_exchange
			}
		}
	}
}
sell_foreign_reserves = {
	effect = {
		# Reduces Reserves
		if = {
			limit = {
				foreign_reserves_value >= 1
			}
			change_variable = {
				name = foreign_reserves
				subtract = foreign_reserves_sell
			}
		}
		if = {
			limit = {
				foreign_reserves_value <= foreign_reserves_sell
			}
			set_variable = {
				name = foreign_reserves
				value = 0
			}
		}
		# Increases Local Currency Value
		on_action_change_domestic_currency_value = {
			mathAction = add
			mathAction2 = currency_value_growth_by_exchange
		}
		add_treasury = 500000
	}
}
### Interest Rate
lower_interest_rate = {
	effect = {
		if = {
			limit = {
				var:interest_rate_gui > 0
			}
			change_variable = {
				name = interest_rate_gui
				subtract = 0.5
			}
		}
		if = {
			limit = {
				var:interest_rate_gui > -100
				var:interest_rate_gui <= 0
			}
			change_variable = {
				name = interest_rate_gui
				subtract = 0.1
			}
		}
	}
}

raise_interest_rate = {
	effect = {
		if = {
			limit = {
				var:interest_rate_gui < 100
				var:interest_rate_gui >= 0
			}
			change_variable = {
				name = interest_rate_gui
				add = 0.5
			}
		}
		if = {
			limit = {
				var:interest_rate_gui > -100
				var:interest_rate_gui < 0
			}
			change_variable = {
				name = interest_rate_gui
				add = 0.1
			}
		}
	}
}

submit_interest_rate = {
	effect = {
		set_variable = {
			name = interest_rate
			value = var:interest_rate_gui
		}
		set_variable = {
			name = changed_interest_rates
			days = 365
		}
		remove_modifier = interest_rates
		add_modifier = {
			name = interest_rates
			multiplier = interest_rate
		}
	}
}

### Monetary Policy

set_contractionary_monetary_policy = {
	effect = {
		remove_variable = stable_monetary_policy
		remove_variable = expansionary_monetary_policy
		set_variable = contractionary_monetary_policy
	}
}

set_stable_monetary_policy = {
	effect = {
		remove_variable = contractionary_monetary_policy
		remove_variable = expansionary_monetary_policy
		set_variable = stable_monetary_policy
	}
}

set_expansionary_monetary_policy = {
	effect = {
		remove_variable = stable_monetary_policy
		remove_variable = contractionary_monetary_policy
		set_variable = expansionary_monetary_policy
	}
}

### Currency
devalue_currency = {
	effect = {
		on_action_change_domestic_currency_value = {
			mathAction = subtract
			mathAction2 = { value = currency_value_growth_by_exchange multiply = 10}
		}
		change_variable = {
			name = money_supply
			subtract = {
				value = money_supply_from_gold_reserves
				divide = 3
			}
		}
	}
}

revalue_currency = {
	effect = {
		on_action_change_domestic_currency_value = {
			mathAction = add
			mathAction2 = { value = currency_value_growth_by_exchange multiply = 10 }
		}
		change_variable = {
			name = money_supply
			add = {
				value = money_supply_from_gold_reserves
				divide = 3
			}
		}
	}
}
