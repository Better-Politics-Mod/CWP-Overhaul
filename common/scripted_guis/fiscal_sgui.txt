### Interest Rate
lower_interest_rate = {
	effect = {
		if = {
			limit = {
				var:interest_rate_gui > 0
			}
			change_variable = {
				name = interest_rate_gui
				subtract = 0.5
			}
		}
		if = {
			limit = {
				var:interest_rate_gui > -100
				var:interest_rate_gui <= 0
			}
			change_variable = {
				name = interest_rate_gui
				subtract = 0.1
			}
		}
	}
}

raise_interest_rate = {
	effect = {
		if = {
			limit = {
				var:interest_rate_gui < 100
				var:interest_rate_gui >= 0
			}
			change_variable = {
				name = interest_rate_gui
				add = 0.5
			}
		}
		if = {
			limit = {
				var:interest_rate_gui > -100
				var:interest_rate_gui < 0
			}
			change_variable = {
				name = interest_rate_gui
				add = 0.1
			}
		}
	}
}

submit_interest_rate = {
	effect = {
		set_variable = {
			name = interest_rate
			value = var:interest_rate_gui
		}
		set_variable = {
			name = changed_interest_rates
			days = 365
		}
		remove_modifier = interest_rates
		add_modifier = {
			name = interest_rates
			multiplier = interest_rate
		}
	}
}

### Fiscal Policy

set_contractionary_fiscal_policy = {
	effect = {
		remove_variable = stable_fiscal_policy
		remove_variable = expansionary_fiscal_policy
		set_variable = contractionary_fiscal_policy
	}
}

set_stable_fiscal_policy = {
	effect = {
		remove_variable = contractionary_fiscal_policy
		remove_variable = expansionary_fiscal_policy
		set_variable = stable_fiscal_policy
	}
}

set_expansionary_fiscal_policy = {
	effect = {
		remove_variable = stable_fiscal_policy
		remove_variable = contractionary_fiscal_policy
		set_variable = expansionary_fiscal_policy
	}
}
