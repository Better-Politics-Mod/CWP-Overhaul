a_new_constitution_button_support_communist = {
	name = a_new_constitution_button_support_communist
	desc = a_new_constitution_button_support_communist_desc
	visible = {
		always = yes
	}
	possible = {
		trigger_if = {
			limit = {
				has_variable = has_selected_communist_once
			}
			custom_tooltip = {
				text = cannot_choose_twice_tt
				always = no
			}
		}
		any_interest_group = {
			is_in_government = yes
			is_interest_group_type = ig_laborists
		}
		custom_tooltip = {
			text = not_already_supporting_communist_tt
			not = {
				has_variable = supporting_communist
			}
		}
	}
	effect = {
		set_variable = supporting_communist
		set_variable = has_selected_communist_once
		remove_variable = supporting_socialist
		remove_variable = supporting_moralist
		remove_variable = supporting_gaullist
		interest_group_ig_laborists_pol_str_mult = 1.5
		custom_tooltip = {
			text = communist_support_increase_1_weekly_tt
			change_variable = {
				name = fra_constitutional_referendum_communist_support
				add = 3
			}
		}
	}
}

a_new_constitution_button_support_gaullist = {
	name = a_new_constitution_button_support_gaullist
	desc = a_new_constitution_button_support_gaullist_desc
	visible = {
		always = yes
	}
	possible = {
		trigger_if = {
			limit = {
				has_variable = has_selected_gaullist_once
			}
			custom_tooltip = {
				text = cannot_choose_twice_tt
				always = no
			}
		}
		any_interest_group = {
			is_in_government = yes
			or = {
				is_interest_group_type = ig_securocrats
				has_ideology = ideology:ideology_gaullist
			}
		}
		custom_tooltip = {
			text = not_already_supporting_gaullist_tt
			not = {
				has_variable = supporting_gaullist
			}
		}
	}
	effect = {
		set_variable = supporting_gaullist
		set_variable = has_selected_gaullist_once
		remove_variable = supporting_communist
		remove_variable = supporting_socialist
		remove_variable = supporting_moralist
		every_interest_group = {
			limit = {
				leader = {
					has_ideology = ideology:ideology_gaullist
				}
			}
			interest_group_pol_str_mult = 1.5
		}
		custom_tooltip = {
			text = gaullist_support_increase_1_weekly_tt
			change_variable = {
				name = fra_constitutional_referendum_gaullist_support
				add = 3
			}
		}
	}
}
