
korean_war_diplo_delegation = {
	name = "korean_war_delegation_button"
	desc = "korean_war_delegation_button_desc"

	visible = {
		OR = {
			has_variable = non_communist_leading
			has_variable = communist_leading
		}
	}
	
	ai_chance = {
		base = 0
	}

	possible = {
		custom_tooltip = {
			text = "We're a leading country in the conflict"
			OR = {
				has_variable = non_communist_leading
				has_variable = communist_leading
			}
		}
	}
	
	effect = {
	}
}

korean_war_auto_decline = {
	name = "korean_war_auto_decline_button"
	desc = "korean_war_auto_decline_button_desc"

	visible = {
		OR = {
			has_variable = non_communist_leading
			has_variable = communist_leading
		}
		NOT = {
			has_variable = auto_decline
		}
		OR = {
			has_variable = ask_for_consent
			has_variable = auto_accept
		}
	}
	
	ai_chance = {
		base = 0
	}

	possible = {
		custom_tooltip = {
			text = "We're a leading country in the conflict"
			OR = {
				has_variable = non_communist_leading
				has_variable = communist_leading
			}
		}
		NOT = {
			has_variable = button_cooldown
		}
	}
	
	effect = {
		if = {
			limit = {
				has_variable = ask_for_consent
			}
			remove_variable = ask_for_consent
		}
		if = {
			limit = {
				has_variable = auto_accept
			}
			remove_variable = auto_accept
		}
		set_variable = auto_decline
		set_variable = {
			name = button_cooldown
			days = 14
		}
	}
}
