# Currency

currency_value_growth = {
	value = 0
	if = {
		limit = {
			money_supply_percentage >= 10
		}
		subtract = {
			value = money_supply_growth_percentage
			divide = 100
		}
	}
	if = {
		limit = {
			money_supply_percentage <= -10
		}
		add = {
			value = money_supply_growth_percentage
			divide = 100
		}
	}
	if = {
		limit = {
			interest_rate >= 1
		}
		add = {
			value = interest_rate
			divide = 2
		}
	}
	if = {
		limit = {
			interest_rate < 1
		}
		subtract = {
			value = interest_rate
			divide = 2
		}
	}
}
