france_constitution_ig_stance_value = {
    value = 1
    if = { limit = { law_stance = {law = law_type:$law$  value = strongly_approve}}
        add = 5
    }
    else_if = { limit = { law_stance = {law = law_type:$law$  value = approve}}
        add = 3
    }
    else_if = { limit = { law_stance = {law = law_type:$law$  value = disapprove}}
        subtract = 1
    }
    else_if = { limit = { law_stance = {law = law_type:$law$  value = strongly_disapprove}}
        subtract = 3
    }
}
