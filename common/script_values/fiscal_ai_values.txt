ai_ideal_interest_rate = {
	value = 2
	add = ai_interest_rate_from_growth
	add = ai_interest_from_inflation
	max = 100
	min = 0
}
ai_interest_from_inflation = {
	if = {
		limit = {
			inflation_value >= 15
		}
		add = 2
	}
	if = {
		limit = {
			inflation_value >= 20
		}
		add = 2
	}
	if = {
		limit = {
			inflation_value >= 25
		}
		add = 2
	}
}

ai_interest_rate_from_growth = {
	value = {
		value = inflation_growth
		add = interest_rate_effect
	}
	divide = {
		value = 0.1
		if = {
			limit = {
				has_law = law_type:law_government_controlled
			}
			divide = 5
		}
		if = {
			limit = {
				has_law = law_type:law_government_interference
			}
			divide = 2.5
		}
		if = {
			limit = {
				has_law = law_type:law_independent_monetary_policy
			}
			multiply = 5
		}
	}
	floor = yes
}
