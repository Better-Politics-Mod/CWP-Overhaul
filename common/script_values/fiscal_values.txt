# Money Supply
money_supply = {
	value = var:money_supply
}

money_supply_percentage = {
	value = money_supply
	divide = money_supply_max
	multiply = 100
}

money_supply_growth_percentage = {
	value = money_supply_growth
	divide = money_supply_max
	multiply = 100
}

money_supply_from_gold_reserves = {
	value = gold_reserves_limit
	divide = 80
}

money_supply_growth_from_gold = {
	value = 0 
	if = {
		limit = {
			gold_reserves > 1
		}
		subtract = money_supply_from_gold_reserves
	}
	if = {
		limit = {
			gold_reserves <= 0
		}
		add = money_supply_from_gold_reserves
	}
}

money_supply_growth = {
	value = 0
	if = {
		limit = {
			has_variable = contractionary_fiscal_policy
		}
		subtract = {
			value = {
				value = gold_reserves_limit
				divide = 70
			}
			desc = MNS_CONTRACTIONARY
			format = "MONEY_SUPPLY_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			has_variable = stable_fiscal_policy
		}
		value = 0
	}
	if = {
		limit = {
			has_variable = expansionary_fiscal_policy
		}
		add = {
			value = {
				value = gold_reserves_limit
				divide = 70
			}
			desc = MNS_CONTRACTIONARY
			format = "MONEY_SUPPLY_TOOLTIP_FORMAT"
		}
	}
	add = {
		value = money_supply_growth_from_gold
		desc = MNS_MONEY_SUPPLY_FROM_RESERVES
		format = "MONEY_SUPPLY_TOOLTIP_FORMAT"
	}
}

### Inflation
inflation_value = {
	value = var:inflation_rate
}

inflation_modifier_mult = {
	value = 0
	if = {
		limit = {
			inflation_value >= 5
			inflation_value < 10
		}
		value = 5
	}
	else_if = {
		limit = {
			inflation_value >= 10
			inflation_value < 15
		}
		value = 15
	}
	else_if = {
		limit = {
			inflation_value >= 15
			inflation_value < 20
		}
		value = 25
	}
	else_if = {
		limit = {
			inflation_value >= 20
			inflation_value < 25
		}
		value = 35
	}
	else_if = {
		limit = {
			inflation_value >= 25
			inflation_value < 30
		}
		value = 50
	}
	else_if = {
		limit = {
			inflation_value >= 30
			inflation_value < 50
		}
		value = 60
	}
	else_if = {
		limit = {
			inflation_value >= 50
			inflation_value < 500
		}
		value = 75
	}
	else_if = {
		limit = {
			inflation_value >= 500
		}
		value = 100
	}
	else_if = {
		limit = {
			inflation_value < 0
			inflation_value >= -5
		}
		value = 5
	}
	else_if = {
		limit = {
			inflation_value < -5
			inflation_value >= -10
		}
		value = 50
	}
	else_if = {
		limit = {
			inflation_value < -10
		}
		value = 100
	}
}

inflation_growth = {
	value = 0
	# From Excess Money Supply
	if = {
		limit = {
			money_supply_percentage >= 10
		}
		add = {
			value = {
				value = money_supply_percentage
				divide = 30
			}
			desc = INF_MONEY_SUPPLY
			format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
		}
	}
	## From Goods prices
	# Housing
	add = {
		value = inflation_housing_price
		desc = INF_HOUSING_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# Groceries
	add = {
		value = inflation_groceries_price
		desc = INF_GROCERIES_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# Clothes
	add = {
		value = inflation_clothes_price
		desc = INF_CLOTHES_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# Services
	add = {
		value = inflation_services_price
		desc = INF_SERVICES_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	#Furniture
	add = {
		value = inflation_furniture_price
		desc = INF_FURNITURE_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# Electricity
	add = {
		value = inflation_electricity_price
		desc = INF_ELECTRICITY_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# Media
	add = {
		value = inflation_media_price
		desc = INF_MEDIA_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# Healthcare
	add = {
		value = inflation_healthcare_price
		desc = INF_HEALTHCARE_PRICE
		format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
	}
	# From Deficit Money Supply
	if = {
		limit = {
			money_supply_percentage <= -10
		}
		add = {
			value = {
				value = money_supply_percentage
				divide = 30
			}
			desc = INF_DEF_MONEY_SUPPLY
			format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
		}
	}
	## From Interest Rate
	if = {
		limit = {
			interest_rate > -100
		}
		subtract = {
			value = {
				value = interest_rate_effect
			}
			desc = INF_INTEREST_RATE
			format = "INFLATION_GROWTH_TOOLTIP_FORMAT"
		}
	}
}
