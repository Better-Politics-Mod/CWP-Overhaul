inflation_modifier_mult = {
	value = 0
	if = {
		limit = {
			inflation_value >= 5
			inflation_value < 10
		}
		value = 5
	}
	else_if = {
		limit = {
			inflation_value >= 10
			inflation_value < 15
		}
		value = 15
	}
	else_if = {
		limit = {
			inflation_value >= 15
			inflation_value < 20
		}
		value = 25
	}
	else_if = {
		limit = {
			inflation_value >= 20
			inflation_value < 25
		}
		value = 35
	}
	else_if = {
		limit = {
			inflation_value >= 25
			inflation_value < 30
		}
		value = 50
	}
	else_if = {
		limit = {
			inflation_value >= 30
			inflation_value < 50
		}
		value = 60
	}
	else_if = {
		limit = {
			inflation_value >= 50
			inflation_value < 500
		}
		value = 75
	}
	else_if = {
		limit = {
			inflation_value >= 500
		}
		value = 100
	}
	else_if = {
		limit = {
			inflation_value < 0
			inflation_value >= -5
		}
		value = 5
	}
	else_if = {
		limit = {
			inflation_value < -5
			inflation_value >= -10
		}
		value = 50
	}
	else_if = {
		limit = {
			inflation_value < -10
		}
		value = 100
	}
}
