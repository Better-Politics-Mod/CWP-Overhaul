# Amicable separation, get customs union
decolonization.2 = {
	type = country_event
	title = decolonization.2.t
	desc = decolonization.2.desc
	flavor = decolonization.2.f
	trigger = {
	}
	event_image = {
		video = "europenorthamerica_london_center"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	duration = 3
	immediate = {
	}
	option = {
		name = decolonization.2.a
		default_option = yes
		add_modifier = {
			name = sun_setting_on_empire
			months = very_long_modifier_time
		}
		every_subject_or_below = {
			limit = {
				is_direct_subject_of = ROOT
				NOT = {
					is_subject_type = subject_type_military_occupation
				}
			}
			make_independent = yes
			ROOT = {
				change_relations = {
					country = prev
					value = 50
				}
			}
		}
		ai_chance = {
			base = 0
		}
	}
	option = {
		name = decolonization.2.b
		every_subject_or_below = {
			limit = {
				is_direct_subject_of = ROOT
				NOT = {
					is_subject_type = subject_type_military_occupation
				}
			}
			make_independent = yes
			ROOT = {
				create_diplomatic_pact = {
					country = prev
					type = customs_union
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
}
