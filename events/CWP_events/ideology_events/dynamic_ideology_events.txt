dynamic_ideologies.1 = {
	type = country_event
	title = dynamic_ideologies.1.t
	desc = dynamic_ideologies.1.d
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = {
		video = "gfx/event_pictures/middleeast_courtroom_upheval.bk2"
		#todo pic
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"
	duration = 3
	trigger = {
		always = yes
	}
	immediate = {
	}
	option = {
		name = dynamic_ideologies.1.a
		default_option = yes
		ig:ig_laborists = {
			if = {
				limit = {
					has_ideology = ideology:ideology_revolutionary
				}
				remove_ideology = ideology_revolutionary
			}
			else_if = {
				limit = {
					has_ideology = ideology:ideology_vanguardist
				}
				remove_ideology = ideology_vanguardist
			}
			else_if = {
				limit = {
					has_ideology = ideology:ideology_socialist
				}
				remove_ideology = ideology_socialist
			}
			add_ideology = ideology_social
		}
		ai_chance = {
			base = 100
		}
	}
}

dynamic_ideologies.9 = {
	type = country_event
	title = dynamic_ideologies.9.t
	desc = dynamic_ideologies.9.d
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = {
		video = "gfx/event_pictures/middleeast_courtroom_upheval.bk2"
		#todo pic
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"
	duration = 3
	trigger = {
		always = yes
	}
	option = {
		name = dynamic_ideologies.9.a
		default_option = yes
		ig:ig_moralists = {
			remove_ideology = ideology_secular
			if = {
				limit = {
					owner.country_has_state_religion = rel:atheist
				}
				set_state_religion = root.country_definition.religion
			}
			if = {
				limit = {
					owner.country_has_state_religion = rel:animist
				}
				add_ideology = ideology_animist
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:mahayana
				}
				add_ideology = ideology_mahayana
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:hindu
				}
				add_ideology = ideology_hindu
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:theravada
				}
				add_ideology = ideology_theravada
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:shiite
				}
				add_ideology = ideology_shiite
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:sunni
				}
				add_ideology = ideology_sunni
			}
			else_if = {
				limit = {
					and = {
						owner.country_has_state_religion = rel:protestant
						owner.capital.region = sr:region_north_america
					}
				}
				add_ideology = ideology_evangelical
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:orthodox_christian
				}
				add_ideology = ideology_orthodox_christian
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:catholic
				}
				add_ideology = ideology_catholic
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:protestant
				}
				add_ideology = ideology_protestant
			}
			else_if = {
				limit = {
					owner.country_has_state_religion = rel:israelite
				}
				add_ideology = ideology_israelite
			}
		}
		ai_chance = {
			base = 100
		}
	}
}
