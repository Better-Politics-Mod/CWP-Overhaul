namespace = ai_fiscal

ai_fiscal.1 = { #Sets Monetary policy
	type = country_event
	hidden = yes
	trigger = {
		is_ai = yes
		not = {
			has_variable = adjusted_monetary_policy
		}
	}
	duration = 3
	immediate = {
		set_variable = {
			name = adjusted_monetary_policy
			days = 30
		}
	}
	option = { # Contractionary
		name = ai_fiscal.1.a
		remove_variable = stable_monetary_policy
		remove_variable = expansionary_monetary_policy
		set_variable = contractionary_monetary_policy
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					money_supply_percentage >= 30
				}
				add = 50
			}
		}
	}
	option = { # Stable
		name = ai_fiscal.2.a
		remove_variable = contractionary_monetary_policy
		remove_variable = expansionary_monetary_policy
		set_variable = stable_monetary_policy
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					money_supply_percentage >= 5
					money_supply_percentage <= 10
				}
				add = 20
			}
		}
	}
	option = { # Expansionary
		name = ai_fiscal.3.a
		remove_variable = stable_monetary_policy
		remove_variable = contractionary_monetary_policy
		set_variable = expansionary_monetary_policy
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					money_supply_percentage <= -30
				}
				add = 50
			}
			modifier = {
				trigger = {
					money_supply_percentage < 5
				}
				add = 50
			}
		}
	}

}

ai_fiscal.2 = { # Credibility Decrease
	type = country_event
	hidden = yes
	trigger = {
		is_ai = yes
		not = {
			has_variable = adjusted_monetary_policy
		}
	}
	duration = 3
	immediate = {
		set_variable = {
			name = previous_bank_credibility
			value = var:bank_credibility
		}
		set_variable = {
			name = bank_credibility
			value = bank_credibility_val
		}
	}
	option = {
		name = cwp_bank_reevaluation.2.a
		default_option = yes
	}

}
