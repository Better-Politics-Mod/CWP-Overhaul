### random events triggered by the JE

#Skirmish on the border
korea_war_events.10 = {
	type = country_event
	placement = scope:korean_border_state

	duration = 3

	title = korea_war_events.10.t
	desc = korea_war_events.10.desc
	flavor = korea_war_events.10.f

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		has_journal_entry = korean_war
		OR = {
			this = c:NRK
			this = c:STK
		}
		NOR = {
			has_global_variable = event_on_korean_border
			has_global_variable = ongoing_skirmish_event
			has_global_variable = skirmish_cooldown
		}
	}
	
	immediate = {
		set_global_variable = event_on_korean_border
		set_global_variable = ongoing_skirmish_event
		set_global_variable = {
			name = skirmish_cooldown
			months = 6
		}
		random_scope_state = {
			limit = {
				owner = c:STK
				any_neighbouring_state = {
					owner = c:NRK
					NOT = {
						owner = root
					}
				}
			}
			save_scope_as = korean_border_state
		}
		random_scope_state = {
			limit = {
				owner = c:NRK
				any_neighbouring_state = {
					owner = c:STK
					NOT = {
						owner = root
					}
				}
			}
			save_scope_as = korean_border_state
		}
	}

	# leader of the conflict wants to be informed about all events
	option = {
		name = korea_war_events.10.a
		default_option = yes


		trigger = {
			OR = {
				AND = {
					this = c:STK
					scope:non_communist_leader = {
						has_variable = ask_for_consent
					}
				}
				AND = {
					this = c:NRK
					scope:communist_leader = {
						has_variable = ask_for_consent
					}
				}
			}
		}
		hidden_effect = {
			remove_global_variable = event_on_korean_border
		}
		if = {
			limit = {
				this = c:STK
			}
			scope:non_communist_leader = {
				trigger_event = {
					id = korea_war_events.100
				}
			}
		}
		else_if = {
			limit = {
				this = c:NRK
			}
			scope:communist_leader = {
				trigger_event = {
					id = korea_war_events.100
				}
			}
		}

		#TODO: AI
		ai_chance = {
			base = 50
		}
	}
	# leader of the conflict automatically declines all propositions
	option = {
		name = korea_war_events.10.b

		hidden_effect = {
			remove_global_variable = event_on_korean_border
		}

		trigger = {
			OR = {
				AND = {
					this = c:STK
					scope:non_communist_leader = {
						has_variable = auto_decline
					}
				}
				AND = {
					this = c:NRK
					scope:communist_leader = {
						has_variable = auto_decline
					}
				}
			}
		}

		#TODO: AI
		ai_chance = {
			base = 0
		}
	}
	# leader of the conflict automatically accepts all propositions
	option = {
		name = korea_war_events.10.c

		hidden_effect = {
			remove_global_variable = event_on_korean_border
		}

		trigger = {
			OR = {
				AND = {
					this = c:STK
					scope:non_communist_leader = {
						has_variable = auto_accept
					}
				}
				AND = {
					this = c:NRK
					scope:communist_leader = {
						has_variable = auto_accept
					}
				}
			}
		}

		if = {
			limit = {
				this = c:STK
			}
			c:NRK = {
				trigger_event = {
					id = korea_war_events.11
				}
			}
		}
		else_if = {
			limit = {
				this = c:NRK
			}
			c:STK = {
				trigger_event = {
					id = korea_war_events.11
				}
			}
		}

		#TODO: AI
		ai_chance = {
			base = 0
		}
	}
	option = {
		name = korea_war_events.10.d
	}
}

#Skirmish on the border, target
korea_war_events.11 = {
	type = country_event
	placement = scope:korean_border_state

	duration = 1

	title = korea_war_events.11.t
	desc = korea_war_events.11.d
	flavor = korea_war_events.11.f

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		has_journal_entry = korean_war
		OR = {
			this = c:NRK
			this = c:STK
		}
		NOT = {
			has_global_variable = event_on_korean_border
		}
	}
	
	immediate = {
		random_scope_state = {
			limit = {
				owner = c:STK
				any_neighbouring_state = {
					owner = c:NRK
					NOT = {
						owner = root
					}
				}
			}
			save_scope_as = korean_border_state
		}
		random_scope_state = {
			limit = {
				owner = c:NRK
				any_neighbouring_state = {
					owner = c:STK
					NOT = {
						owner = root
					}
				}
			}
			save_scope_as = korean_border_state
		}
	}

	option = {
		name = korea_war_events.11.a
		default_option = yes

		trigger = {
			global_var:korean_tensions < 90
		}

		change_global_variable = {
			name = korean_tensions
			add = {
				integer_range = {
					min = 1
					max = 5
				}
			}
		}

		hidden_effect = {
			remove_global_variable = ongoing_skirmish_event
		}

		#TODO: AI
		ai_chance = {
			base = 50
		}
	}
	option = {
		name = korea_war_events.11.b

		change_global_variable = {
			name = korean_tensions
			subtract = {
				integer_range = {
					min = 1
					max = 5
				}
			}
		}

		hidden_effect = {
			remove_global_variable = event_on_korean_border
		}
		hidden_effect = {
			remove_global_variable = ongoing_skirmish_event
		}

		#TODO: AI
		ai_chance = {
			base = 0
		}
	}
}

#consent event
korea_war_events.100 = {
	type = country_event
	placement = ROOT

	duration = 1

	title = korea_war_events.100.t
	desc = korea_war_events.100.d
	flavor = korea_war_events.100.f

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		has_journal_entry = korean_war
		OR = {
			has_variable = communist_leading
			has_variable = non_communist_leading
		}
	}
	
	immediate = {
		save_scope_as = originator
	}

	option = {
		name = korea_war_events.100.a
		default_option = yes

		trigger = {
			has_global_variable = ongoing_skirmish_event
		}

		if = {
			limit = {
				this = scope:non_communist_leader
			}
			c:NRK = {
				trigger_event = {
					id = korea_war_events.11
				}
			}
			c:STK = {
				post_notification = leader_agreed
			}
		}
		else_if = {
			limit = {
				this = scope:communist_leader
			}
			c:STK = {
				trigger_event = {
					id = korea_war_events.11
				}
			}
			c:NRK = {
				post_notification = leader_agreed
			}
		}

		#TODO: AI
		ai_chance = {
			base = 50
		}
	}
	option = {
		name = korea_war_events.100.b

		hidden_effect = {
			remove_global_variable = event_on_korean_border
		}
		hidden_effect = {
			remove_global_variable = ongoing_skirmish_event
		}

		if = {
			limit = {
				this = scope:non_communist_leader
			}
			c:STK = {
				post_notification = leader_disagreed
			}
		}
		else_if = {
			limit = {
				this = scope:communist_leader
			}
			c:NRK = {
				post_notification = leader_disagreed
			}
		}

		#TODO: AI
		ai_chance = {
			base = 0
		}
	}
}
